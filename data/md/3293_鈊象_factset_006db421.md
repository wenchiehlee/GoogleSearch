<script>
(function() {
  'use strict';
  
  // DOM 元素 (使用 header-search-hint 前綴避免衝突)
  const headerSearchInput = document.querySelector('.header-search__input');
  const headerSearchHint = document.querySelector('.header-search-hint');
  const headerHistoryListContainer = document.querySelector('.header-search-hint__history-list');
  const headerHistoryEmptyMsg = document.querySelector('.header-search-hint__empty');
  const headerClearAllHistoryBtn = document.querySelector('.header-search-hint__action');
  const headerHotKeywords = document.querySelectorAll('.header-search-hint__item');
  const MAX_HISTORY_COUNT = 5;
  
  // 顯示搜尋提示
  function showHeaderSearchHint() {
    if (headerSearchHint) {
      headerSearchHint.classList.add('is-active');
    }
  }
  
  // 隱藏搜尋提示
  function hideHeaderSearchHint() {
    if (headerSearchHint) {
      headerSearchHint.classList.remove('is-active');
    }
  }
  
  // HTML 字元實體編碼 (HTML Entity Encoding) - XSS 防護
  function escapeHtml(str) {
    if (!str || typeof str !== 'string') return '';
    
    // 使用字元替代方式處理特殊字元
    const htmlEntities = {
      '&': '&amp;',
      '<': '&lt;',
      '>': '&gt;',
      '"': '&quot;',
      "'": '&#x27;',
      '/': '&#x2F;'
    };
    
    return str.replace(/[&<>"'\/]/g, function(match) {
      return htmlEntities[match];
    });
  }
  
  // 解碼 HTML 實體（用於顯示）
  function unescapeHtml(str) {
    if (!str || typeof str !== 'string') return '';
    
    const htmlEntities = {
      '&amp;': '&',
      '&lt;': '<',
      '&gt;': '>',
      '&quot;': '"',
      '&#x27;': "'",
      '&#x2F;': '/'
    };
    
    return str.replace(/&amp;|&lt;|&gt;|&quot;|&#x27;|&#x2F;/g, function(match) {
      return htmlEntities[match];
    });
  }
  
  // XSS 安全性檢查函數（檢測可疑模式）
  function containsXSSPattern(str) {
    if (!str || typeof str !== 'string') return false;
    
    const xssPatterns = [
      /<script/i,
      /<\/script>/i,
      /<iframe/i,
      /javascript:/i,
      /on\w+\s*=/i,
      /<img[^>]*onerror/i,
      /<svg[^>]*onload/i,
      /eval\s*\(/i,
      /expression\s*\(/i,
      /<embed/i,
      /<object/i,
      /vbscript:/i,
      /data:text\/html/i,
      /<style/i,
      /<link/i,
      /<meta/i,
      /<base/i,
    ];
    
    const hasXSS = xssPatterns.some(pattern => pattern.test(str));
    const hasHtmlTags = /<[a-z][\s\S]*>/i.test(str);
    
    return hasXSS || hasHtmlTags;
  }
  
  // 安全地創建歷史記錄項目（使用 DOM API + HTML 實體解碼）
  function createHeaderHistoryItem(keyword) {
    if (containsXSSPattern(keyword)) {
      console.warn('檢測到可疑內容，已阻止:', keyword);
      return null;
    }
    
    const item = document.createElement('div');
    item.className = 'header-search-hint__history-item';
    
    // 解碼後的關鍵字用於顯示和跳轉
    const decodedKeyword = unescapeHtml(keyword);
    item.dataset.history = decodedKeyword;
    
    const icon = document.createElement('i');
    icon.className = 'i-time';
    
    const text = document.createElement('span');
    text.className = 'header-search-hint__history-text';
    text.textContent = decodedKeyword; // 使用 textContent 防止 XSS
    
    const removeBtn = document.createElement('button');
    removeBtn.type = 'button';
    removeBtn.className = 'header-search-hint__history-remove';
    
    const closeIcon = document.createElement('i');
    closeIcon.className = 'i-close';
    removeBtn.appendChild(closeIcon);
    
    item.appendChild(icon);
    item.appendChild(text);
    item.appendChild(removeBtn);
    
    return item;
  }
  
  // 渲染歷史列表
  function renderHeaderHistoryList() {
    let historyData = localStorage.getItem('search_history');
    
    if (!historyData) {
      headerHistoryListContainer.style.display = 'none';
      headerHistoryEmptyMsg.style.display = 'block';
      return;
    }
    
    try {
      const parsed = JSON.parse(historyData);
      const historyArray = parsed.search_history || [];
      
      if (historyArray.length === 0) {
        headerHistoryListContainer.style.display = 'none';
        headerHistoryEmptyMsg.style.display = 'block';
        return;
      }
      
      headerHistoryEmptyMsg.style.display = 'none';
      headerHistoryListContainer.style.display = 'block';
      headerHistoryListContainer.innerHTML = '';
      
      historyArray.forEach((keyword) => {
        const itemElement = createHeaderHistoryItem(keyword);
        if (itemElement) {
          headerHistoryListContainer.appendChild(itemElement);
        }
      });
      
      bindHeaderRemoveEvents();
      bindHeaderHistoryItemEvents();
    } catch (e) {
      console.error('解析搜尋歷史失敗:', e);
      headerHistoryListContainer.style.display = 'none';
      headerHistoryEmptyMsg.style.display = 'block';
    }
  }
  
  // 綁定刪除單筆歷史事件
  function bindHeaderRemoveEvents() {
    const removeButtons = headerHistoryListContainer.querySelectorAll('.header-search-hint__history-remove');
    removeButtons.forEach(btn => {
      btn.addEventListener('click', function (e) {
        e.stopPropagation();
        const item = this.closest('.header-search-hint__history-item');
        const keyword = item.dataset.history;
        removeHeaderHistoryItem(keyword);
      });
    });
  }
  
  // 綁定歷史記錄點擊事件
  function bindHeaderHistoryItemEvents() {
    const historyItems = document.querySelectorAll('.header-search-hint__history-item');
    historyItems.forEach(item => {
      item.addEventListener('click', function (e) {
        if (e.target.closest('.header-search-hint__history-remove')) {
          return;
        }
        const keyword = this.dataset.history;
        
        if (containsXSSPattern(keyword)) {
          alert('偵測到可疑的內容，無法執行搜尋。\n請檢查您的輸入資料。');
          console.warn('XSS 攻擊嘗試被阻止 (跳轉時):', keyword);
          return;
        }
        
        window.location.href = `/search/result/1001/${encodeURIComponent(keyword)}`;
      });
    });
  }
  
  // 刪除單筆歷史
  function removeHeaderHistoryItem(keyword) {
    const historyData = localStorage.getItem('search_history');
    if (!historyData) return;
    
    try {
      const parsed = JSON.parse(historyData);
      const historyArray = parsed.search_history || [];
      
      // 編碼關鍵字以匹配儲存的格式
      const encodedKeyword = escapeHtml(keyword);
      
      // 過濾掉指定的關鍵字（同時匹配編碼和未編碼版本）
      const newArray = historyArray.filter(item => 
        item !== keyword && item !== encodedKeyword
      );
      
      localStorage.setItem('search_history', JSON.stringify({
        search_history: newArray
      }));
      
      renderHeaderHistoryList();
    } catch (e) {
      console.error('刪除歷史失敗:', e);
    }
  }
  
  // 新增搜尋歷史
  function addHeaderSearchHistory(keyword) {
    if (!keyword || keyword.trim().length === 0) return;
    
    keyword = keyword.trim();
    
    if (containsXSSPattern(keyword)) {
      alert('偵測到可疑的輸入內容，請檢查您的搜尋關鍵字。\n為了您的安全，此內容不會被記錄。');
      console.warn('XSS 攻擊嘗試被阻止:', keyword);
      return;
    }
    
    if (keyword.length > 100) {
      alert('搜尋關鍵字過長，請輸入 100 字元以內的內容。');
      return;
    }
    
    try {
      let historyData = localStorage.getItem('search_history');
      let historyArray = [];
      
      if (historyData) {
        const parsed = JSON.parse(historyData);
        historyArray = parsed.search_history || [];
      }
      
      // 對關鍵字進行 HTML 實體編碼後再儲存
      const sanitizedKeyword = escapeHtml(keyword);
      
      // 如果已存在相同關鍵字，先移除（避免重複）
      historyArray = historyArray.filter(item => item !== sanitizedKeyword);
      
      // 將新關鍵字加到陣列開頭（最新）
      historyArray.unshift(sanitizedKeyword);
      
      if (historyArray.length > MAX_HISTORY_COUNT) {
        historyArray = historyArray.slice(0, MAX_HISTORY_COUNT);
      }
      
      localStorage.setItem('search_history', JSON.stringify({
        search_history: historyArray
      }));
      
      console.log('搜尋歷史已更新 (已編碼):', historyArray);
    } catch (e) {
      console.error('新增搜尋歷史失敗:', e);
    }
  }
  
  // 清空全部歷史
  function clearHeaderAllHistory() {
    localStorage.removeItem('search_history');
    renderHeaderHistoryList();
  }
  
  // 綁定清空全部按鈕
  if (headerClearAllHistoryBtn) {
    headerClearAllHistoryBtn.addEventListener('click', clearHeaderAllHistory);
  }
  
  // 綁定熱門關鍵字點擊事件
  headerHotKeywords.forEach(item => {
    item.addEventListener('click', function (e) {
      const keyword = this.textContent.trim();
      
      if (containsXSSPattern(keyword)) {
        e.preventDefault();
        alert('偵測到可疑的內容，無法執行搜尋。\n請檢查您的輸入資料。');
        console.warn('XSS 攻擊嘗試被阻止 (熱門關鍵字):', keyword);
        return;
      }
      
      addHeaderSearchHistory(keyword);
      console.log('熱門關鍵字已加入歷史:', keyword);
    });
  });
  
  // 綁定搜尋框 focus/blur 事件
  if (headerSearchInput) {
    headerSearchInput.addEventListener('focus', function() {
      showHeaderSearchHint();
      renderHeaderHistoryList(); // 每次 focus 時重新渲染歷史
    });
    
    headerSearchInput.addEventListener('blur', function() {
      // 延遲隱藏，讓點擊提示項目有時間觸發
      setTimeout(hideHeaderSearchHint, 200);
    });
  }
  
  // 點擊搜尋提示區域時，阻止 input blur
  if (headerSearchHint) {
    headerSearchHint.addEventListener('mousedown', function(e) {
      e.preventDefault(); // 防止 input 失去焦點
    });
    
    headerSearchHint.addEventListener('click', function(e) {
      // 如果點擊的是連結或歷史項目，讓它正常跳轉
      if (e.target.closest('.header-search-hint__item') || 
          e.target.closest('.header-search-hint__history-item')) {
        // 不阻止預設行為，讓跳轉發生
      }
    });
  }
  
  // 頁面載入時渲染
  renderHeaderHistoryList();
  
})();
</script>

function createDivRemover(targetSelector, callback = null) {
    // Create the mutation observer
    const observer = new MutationObserver((mutations) => {
        mutations.forEach((mutation) => {
            // Check if nodes were added
            if (mutation.type === 'childList' && mutation.addedNodes.length > 0) {
                mutation.addedNodes.forEach((node) => {
                    // Check if the added node is an element
                    if (node.nodeType === Node.ELEMENT_NODE) {
                        // Check if the node matches our target selector
                        if (node.matches && node.matches(targetSelector)) {
                            console.log('Target div found:', node);
                            
                            // Execute callback if provided
                            if (callback && typeof callback === 'function') {
                                callback(node);
                            }
                            
                            // Remove the element
                            node.remove();
                            console.log('Target div removed');
                        }
                        
                        // Also check children of the added node
                        const targetElements = node.querySelectorAll ? node.querySelectorAll(targetSelector) : [];
                        targetElements.forEach((targetElement) => {
                            console.log('Target div found in children:', targetElement);
                            
                            // Execute callback if provided

  $(document).ready(function(){
    function hashScroll() {
      target = $(window.location.hash);
      event.preventDefault();
          $('html, body').animate({
              scrollTop: target.offset().top - 50
          }, 500);

      if(window.location.hash.length > 0){
        $("div.money-menu").css("display", "none");
      }
      $("div.money-menu").css("display", "none");
      setTimeout(function() {
      $( window ).scroll(function() {
        $("div.money-menu").css("display", "flex");
      });
      }, 1000);
    }
    document.onreadystatechange = subSomething;//當頁面載入狀態改變的時候執行這個方法.
    function subSomething() {
      if(document.readyState == 'complete') {
        if(window.location.hash.length > 0){
        $("div.money-menu").css("display", "none");
      }
        if(window.location.hash.length > 0) {
          hashScroll()
        }
      } //當頁面載入狀態
    }
    window.addEventListener('hashchange', function() {
        hashScroll()
    }, false);
  });

</script>
